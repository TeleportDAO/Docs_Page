InstantTransfer#

event InstantTransfer(address indexed owner, address indexed spender, uint value);

Emitted each time an instant transfer takes place.

State-Changing Functions#

addLiquidity#

function addLiquidity(uint amount) external returns(bool);

Adds amount of wrapped bitcoin to instant pool.

removeLiquidity#

function removeLiquidity(uint amount) external returns(bool);

Burns amount of instant pool token of msg.sender.

instantCCTransferWithPermit#

function instantCCTransferWithPermit(    address signer,    bytes memory signature,    address receiver,    uint amount,    uint nonce) external returns(bool);

Transfer amount of wrapped bitcoin from instant pool to receiver.

  • signer is the address of the user who wants to perform instant transfer.
  • signature is the signature of the signer under the request data.
  • deadline is the last block number that the request can be executed in it.

instantCCExchangeWithPermit#

function instantCCExchangeWithPermit(    address signer,    bytes memory signature,    uint amountIn,    uint amountOutMin,    address[] memory path,    address receiver,    uint deadline) external returns(bool);

Transfer amountIn of wrapped bitcoin from instant pool to receiver and then calls exchange router to perfom swap.

payBackInstantTransfer#

function payBackInstantTransfer(    uint amount,    uint256 keeperFee,    bytes32 messageHash) external returns(bool);

Pays back the instant loan of request[messageHash].

lockHGT#

function lockHGT(uint amount) external returns(bool);

Locks amount of HGT in instant pool router.

unlockHGT#

function unlockHGT(uint amount) external returns(bool);

Unlocks amount of HGT from instant pool router and sends them back to msg.sender.

punishUser#

function punishUser(bytes32 messageHash) external returns(bool);

Punishes request[messageHash].user if user has not paid back the instant loan before deadline.

setExchangeRouter#

function setExchangeRouter(address _ExchangeRouter) external;

function setDeadline(uint _paybackDeadline) external;
